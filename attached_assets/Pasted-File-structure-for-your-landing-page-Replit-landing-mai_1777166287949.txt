import os, json, base64, re
import anthropic
_client = None
def _get_client():
    global _client
    if _client is None:
        _client = anthropic.Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
    return _client
SYSTEM_PROMPT = """\
You are an expert logistics document analyzer specializing in CMR (International Consignment Notes). \
Your goal is to extract handwritten and printed text from images with maximum accuracy.
CRITICAL INSTRUCTIONS:
- Do not be overly cautious. If you can make a highly probable guess at a handwritten word, provide that guess.
- Contextual Clues: Use the surrounding printed text (e.g., "Sender", "Consignee", "Gross weight") to identify which box you are reading.
- Handwriting Focus: Prioritize the handwritten ink over the printed form lines.
- Formatting: Return the data in strict JSON. If a field is completely blank or unreadable, only then use "Not legible". Otherwise, provide the most likely text.
- Specific Fields to hunt for:
  * Field 1:  Sender name and address (top left).
  * Field 2:  Consignee/receiver name and address (middle left).
  * Field 16: Carrier name and address — look for the box labeled "Carrier" or "Vežėjas" (usually upper-right area).
  * Field 17: Successive/following carriers — look for "Successive carriers" label.
  * Driver:   Look for "Naam Chauffeur", "Driver", "Vairuotojas", or a signature block with a name.
  * Field 25: Truck plate / vehicle registration — look for plate-like text (e.g. "KDF 056").
  * Field 3:  Place of delivery.
  * Field 4:  Place and date of taking over / pickup.
  * Field 6/9: Quantity and item descriptions.
  * Field 11: Gross weight in kg.
  * Field 12: Volume in m³.
  * Field 21: Date of issue AND any Incoterms/shipping terms (e.g. "DAP Genoa", "EXW", "CIF").
  * Stamps:   CRITICALLY look for any red or blue rectangular/oval stamps. If a stamp contains a date, extract it as stamp_date.
  * Notes/contacts: Look in any free-text boxes or margins for ship names, contact persons, phone numbers, or email addresses.\
"""
USER_PROMPT = """\
Extract all CMR fields from this document image and return ONLY a JSON object. \
No text before or after the JSON. No markdown code fences.
Return this exact structure:
{
  "sender":             { "name": "...", "address": "...", "country": "..." },
  "consignee":          { "name": "...", "address": "...", "country": "..." },
  "carrier_name":       "...",
  "following_carriers": "...",
  "place_of_delivery":  "...",
  "pickup_location":    "...",
  "pickup_date":        "...",
  "goods_description":  "...",
  "gross_weight_kg":    "...",
  "volume_m3":          "...",
  "driver_name":        "...",
  "truck_plate":        "...",
  "shipping_terms":     "...",
  "document_date":      "...",
  "stamp_date":         "...",
  "ship_name":          "...",
  "contact_person":     "...",
  "contact_info":       "..."
}
"""
def extract_cmr_fields(image_bytes: bytes, media_type: str = "image/jpeg") -> dict:
    client = _get_client()
    b64 = base64.standard_b64encode(image_bytes).decode()
    msg = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=1500,
        system=SYSTEM_PROMPT,
        messages=[{"role": "user", "content": [
            {"type": "image", "source": {"type": "base64", "media_type": media_type, "data": b64}},
            {"type": "text",  "text": USER_PROMPT}
        ]}]
    )
    raw = msg.content[0].text.strip()
    m = re.search(r"\{.*\}", raw, re.DOTALL)
    if m:
        raw = m.group(0)
    try:
        return json.loads(raw)
    except Exception:
        return {"_error": "Could not parse response", "_raw": raw[:400]}

import os, time
from flask import Flask, request, jsonify, render_template
from flask_cors import CORS
from extractor import extract_cmr_fields
app = Flask(__name__)
CORS(app, resources={r"/api/*": {"origins": "*"}})
# Simple in-memory rate limit: 1 free scan per IP
_used = {}   # ip -> timestamp
COOLDOWN = 86400  # 24 hours in seconds
@app.route("/")
def index():
    return render_template("index.html")
@app.route("/api/free-scan", methods=["POST", "OPTIONS"])
def free_scan():
    if request.method == "OPTIONS":
        return "", 204
    ip = request.headers.get("X-Forwarded-For", request.remote_addr).split(",")[0].strip()
    now = time.time()
    if ip in _used and (now - _used[ip]) < COOLDOWN:
        remaining = int(COOLDOWN - (now - _used[ip])) // 3600
        return jsonify({"error": f"One free scan per day. Try again in ~{remaining}h."}), 429
    f = request.files.get("file")
    if not f:
        return jsonify({"error": "No file uploaded."}), 400
    mime = f.content_type or "image/jpeg"
    allowed = ("image/jpeg", "image/png", "image/webp", "image/gif", "image/tiff")
    if mime not in allowed:
        return jsonify({"error": "Please upload a JPEG, PNG, or WebP image."}), 400
    data = f.read()
    if len(data) > 12 * 1024 * 1024:
        return jsonify({"error": "File too large. Max 12 MB."}), 400
    try:
        result = extract_cmr_fields(data, mime)
        _used[ip] = now
        return jsonify({"ok": True, "data": result})
    except Exception as e:
        return jsonify({"error": str(e)}), 500
if __name__ == "__main__":
    port = int(os.environ.get("PORT", 3000))
    app.run(host="0.0.0.0", port=port)

const HIGHLIGHT_FIELDS = ['truck_plate','driver_name','gross_weight_kg','document_date'];
let selectedFile = null;
// Drag & drop
const dz = document.getElementById('dropZone');
dz.addEventListener('dragover',  e => { e.preventDefault(); dz.classList.add('drag-over'); });

function handleFile(file) {
  selectedFile = file;
  const thumb = document.getElementById('previewThumb');
  thumb.src = URL.createObjectURL(file);
  thumb.style.display = 'block';
  document.getElementById('scanBtn').style.display = 'block';
  document.getElementById('resultsCard').style.display = 'none';
  document.getElementById('errorBanner').style.display = 'none';
}
async function runScan() {
  if (!selectedFile) return;
  const btn    = document.getElementById('scanBtn');
  const status = document.getElementById('statusLine');
  const err    = document.getElementById('errorBanner');
  const result = document.getElementById('resultsCard');
  btn.disabled = true;
  status.style.display = 'flex';
  err.style.display    = 'none';
  result.style.display = 'none';
  document.getElementById('statusText').textContent = 'Sending to Claude AI…';
  const fd = new FormData();
  fd.append('file', selectedFile);
  try {
    // 👇 Change this URL to your deployed Flask backend
    const res  = await fetch('/api/free-scan', { method: 'POST', body: fd });
    const json = await res.json();
    if (!res.ok || json.error) {
      throw new Error(json.error || 'Something went wrong.');
    }
    renderResults(json.data);
    result.style.display = 'block';
  } catch (e) {
    err.textContent      = e.message;
    err.style.display    = 'block';
  } finally {
    btn.disabled         = false;
    status.style.display = 'none';
  }
}
function renderResults(data) {
  const grid = document.getElementById('rcGrid');
  grid.innerHTML = '';
  for (const [key, label] of Object.entries(FIELD_LABELS)) {
    let raw = data[key];
    let display = '';
    if (raw && typeof raw === 'object') {
      // sender / consignee are objects
      display = [raw.name, raw.address, raw.country].filter(Boolean).join(', ');
    } else {
      display = raw || '';
    }
    const isEmpty   = !display || display === 'Not legible';
    const isHighlight = HIGHLIGHT_FIELDS.includes(key) && !isEmpty;
    grid.insertAdjacentHTML('beforeend', `
      <div class="rc-field">
        <div class="rc-label">${label}</div>
        <div class="rc-value ${isHighlight ? 'highlight' : ''} ${isEmpty ? 'empty' : ''}">
          ${isEmpty ? '—' : escHtml(display)}
        </div>
      </div>
    `);
  }
}
function escHtml(s) {
  return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}
